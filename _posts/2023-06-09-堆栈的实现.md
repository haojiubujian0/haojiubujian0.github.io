# 堆栈

- **LIFO**（Last In First Out）：将一个元素添加到堆栈或从堆栈中删除的顺序可以描述为后进先出。
- 两种主要的操作：
  - Push：向堆栈中加入一个元素。
  - Pop：从堆栈中移除最近添加且尚未移除的元素。

- 其他一些操作：
  - "top of stack" or "peak"：访问栈顶元素但不移除该元素。
  - "empty"：判断堆栈是否为空。
  - "size"：返回堆栈的当前大小。

# 应用

- 回溯（Backtracking）
  - 一个典型的回溯算法是深度优先搜索
  - 走迷宫：在一个迷宫中寻找正确路径
- 表达式求值

# 实现

- 堆栈的抽象基类

```c++
// stack.h
#ifndef MY_STACK_H
#define MY_STACK_H
// 栈的抽象基类
template<typename T>
class stack {
public:
    virtual void push(const T& x) = 0;  // 入栈操作
    virtual T pop() = 0;    // 出栈操作
    virtual T& top() = 0;   // 访问栈顶元素
    virtual const T& top() const = 0;  
    virtual int size() const = 0;    // 返回栈内的元素数目
    virtual bool empty() const = 0;     // 栈是否为空
    virtual ~stack() {};
};
#endif
```

- 以动态数组的方式实现

```c++
// seqstack.h
#ifndef MY_SEQSTACK_H
#define MY_SEQSTACK_H
#include <stdexcept>
#include <iostream>

// 前置声明
template<typename> class SeqStack;
template<typename T>
std::ostream& operator<<(std::ostream& os, const SeqStack<T>& stk);

// 采用动态数组实现的堆栈
template<typename T>
class SeqStack {
    friend std::ostream& operator<<<T>(std::ostream& os, const SeqStack<T>& stk);
public:
    SeqStack(int sz = 2) {
        if(sz <= 0) throw std::runtime_error("bad initial size!");
        maxsize = sz;
        data = new T[sz];
        cursize = 0;
    }
    void push(const T& x) {
        if(cursize == maxsize) {
            T *old = data;
            data = new T[2*maxsize];  // 栈空间不足时以两倍的量进行扩容
            for(int i = 0; i < maxsize; ++i) data[i] = old[i];
            maxsize *= 2;
        }
        data[cursize++] = x;
    } 
    T pop() {
        if(cursize == 0) throw std::runtime_error("empty stack!");
        return data[--cursize];
    }
    T& top() {
        if(cursize == 0) throw std::runtime_error("empty stack!");
        return data[cursize-1];
    }
    const T& top() const {
        if(cursize == 0) throw std::runtime_error("empty stack!");
        return data[cursize-1];
    }
    int size() const { return cursize; }
    bool empty() const { return cursize == 0; }

private:
    T *data;    
    int maxsize;
    int cursize;  // 记录当前栈的大小
};

template<typename T>
std::ostream& operator<<(std::ostream& os, const SeqStack<T>& stk)
{
    os << "stack [-> top]: ";
    int n = stk.size();
    for(int i = 0; i < n; ++i) {
        os << stk.data[i] << ' ';
    }
    return os;
}
#endif
```

- 以单链表的方式实现

```c++
#ifndef MY_LINKSTACK_H
#define MY_LINKSTACK_H
#include <stdexcept>

// 前置声明
template<typename> class LinkStack;  // operator<<的参数所需要的
template<typename T>
std::ostream& operator<<(std::ostream& os, const LinkStack<T>& stk);

// 采用链表实现堆栈
template<typename T>
class LinkStack {
    friend std::ostream& operator<<<T>(std::ostream& os, const LinkStack<T>& stk);
    // 链表节点定义
    struct Node {
        T data;
        Node *next;
        Node(const T& d = T(), Node *nt = nullptr): data(d), next(nt) {} 
    };
public:
    LinkStack(): head(nullptr), cursize(0) {}
    void push(const T& x) {
        // 构造新节点
        Node *newhead = new Node(x, head);
        head = newhead;
        ++cursize;
    }
    T pop() {
        // 栈空时
        if(head == nullptr) throw std::runtime_error("empty stack!");
        Node* oldhead = head;
        T r = oldhead->data;
        head = head->next;
        delete oldhead;
        --cursize;
        return r;
    }
    T& top() {
        // 栈空时
        if(head == nullptr) throw std::runtime_error("empty stack!");
        return head->data;
    }
    const T& top() const {
        // 栈空时
        if(head == nullptr) throw std::runtime_error("empty stack!");
        return head->data;
    }
    int size() const { return cursize; }
    bool empty() const { return head == nullptr; }

private:
    Node* head;
    int cursize;
};

template<typename T>
std::ostream& operator<<(std::ostream& os, const LinkStack<T>& stk)
{
    os << "stack [top <-]: ";
    // 默认情况下，c++中假定通过作用域运算符访问的名字不是类型。
    // 这里我们通过关键字typename显式告诉编译器该名字是一个类型。
    typename LinkStack<T>::Node *p = stk.head;
    while(p) {
        os << p->data << ' ';
        p = p->next;
    }
    return os;
}
#endif
```

# 测试

下面是一个简单的测试代码。

```c++
#include <iostream>
#include <vector>
#include "seqstack.h"
#include "linkstack.h"

int main()
{
    SeqStack<int> seqstk;
    LinkStack<int> linkstk;
    std::vector<int> arr;
    // 输入数据
    int x;
    while(std::cin >> x) {
        arr.push_back(x);
    }

    // -----------------------------------------------------
    // 测试SeqStack
    // -----------------------------------------------------
    int popelem, topelem;
    std::cout << "test SeqStack<int>...\n";
    std::cout << "push all elements.\n";
    for(int i = 0; i < arr.size(); ++i) {
        seqstk.push(arr[i]);
        std::cout << seqstk << '\n';
        std::cout << "current stack size: " << seqstk.size() << '\n';
    }
    std::cout << "pop all elements.\n";
    while(!seqstk.empty()) {
        std::cout << "current stack size: " << seqstk.size() << '\n';
        topelem = seqstk.top();
        popelem = seqstk.pop();
        std::cout << "top elem: " << topelem << " pop elem: " << popelem << '\n';
        std::cout << "after pop: \n";
        std::cout << seqstk << '\n';
    }

    // -----------------------------------------------------
    // 测试LinkStack
    // -----------------------------------------------------
    std::cout << "test LinkStack<int>...\n";
    std::cout << "push all elements.\n";
    for(int i = 0; i < arr.size(); ++i) {
        linkstk.push(arr[i]);
        std::cout << linkstk << '\n';
        std::cout << "current stack size: " << linkstk.size() << '\n';
    }
    std::cout << "pop all elements.\n";
    while(!linkstk.empty()) {
        std::cout << "current stack size: " << linkstk.size() << '\n';
        topelem = linkstk.top();
        popelem = linkstk.pop();
        std::cout << "top elem: " << topelem << " pop elem: " << popelem << '\n';
        std::cout << "after pop: \n";
        std::cout << linkstk << '\n';
    }

    system("pause");
}
```
